async function sendDataToAI(data, promptText) { const apiKey = getApiKey(); if (!apiKey) return "Sin API Key."; document.getElementById('spinner').style.display = 'inline-block'; document.getElementById('send-to-ia').disabled = true; // Cambiamos a v1beta que suele ser más flexible con los nombres de modelos flash const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`; try { const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: `Analiza estos datos de Grist:\n${JSON.stringify(data)}\n\nInstrucción: ${promptText}` }] }], generationConfig: { temperature: 0.7, maxOutputTokens: 2048, } }) }); const json = await response.json(); if (json.error) { // Si el error es de modelo no encontrado, intentamos dar una pista if (json.error.message.includes("not found")) { throw new Error("El modelo Gemini 1.5 Flash no fue encontrado en esta región o versión de API."); } throw new Error(json.error.message); } if (json.candidates && json.candidates[0].content) { return json.candidates[0].content.parts[0].text; } else { return "La IA no pudo procesar los datos. Intenta con un prompt más específico."; } } catch (error) { console.error("Error de Gemini:", error); return `Error: ${error.message}`; } finally { document.getElementById('spinner').style.display = 'none'; document.getElementById('send-to-ia').disabled = false; } }
